import java.io.*;
class node
{
int key;
node left,right;
node(int value)
{
key=value;
left=null;
right=null;
}
}
class binarytree
{
node root;
int nodes;
bstree()
{
root=null;
nodes=0;
}
node insert(int value,node root)
{
if(root==null)
return new node(value);
else if(root.left==null && root.right==null)
root.left=insert(value,root.left);
else if(root.left!=null && root.right==null)
root.right=insert(value,root.right);

}
